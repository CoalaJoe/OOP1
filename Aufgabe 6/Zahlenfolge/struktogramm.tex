\documentclass{article}

\usepackage{struktex}

\begin{document}

\begin{struktogramm}(100,40)
    \assign{Startwert einlesen}
    \assign{Endwert einlesen}
    \assign{Schrittweite einlesen}
    \assign{Schrittwert = Startwert}
    \while{Solange Startwert kleiner als Endwert}
        \assign{Schrittwert ausgaben}
        \assign{Startwert += Schrittwert}
    \whileend
\end{struktogramm}

\end{document}
